Quick JS: Find nearest parking slot

// Very small example: choose nearest free slot (Euclidean)
function nearestSlot(current, slots){
  let best = null; 
  let bestD = Infinity;

  for(const s of slots){
    if(!s.free) continue;

    const dx = s.x - current.x; 
    const dy = s.y - current.y;
    const d = Math.sqrt(dx*dx + dy*dy);

    if(d < bestD){
      bestD = d; 
      best = s;
    }
  }

  return best;
}

Quick Python: Dijkstra (for routes)

# Simple Dijkstra (adj list)
import heapq

def dijkstra(adj, start):
    dist = {v: float('inf') for v in adj}
    dist[start] = 0
    pq = [(0, start)]

    while pq:
        d, u = heapq.heappop(pq)
        if d > dist[u]: 
            continue

        for v, w in adj[u]:
            nd = d + w
            if nd < dist[v]:
                dist[v] = nd
                heapq.heappush(pq, (nd, v))

    return dist
